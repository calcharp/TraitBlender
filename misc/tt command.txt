def register_to_config(name: str):
	"""This decorator takes a PropertyGroup and adds it to dict that gets converted to the	 	    global config registry"""
	def decorator(cls: PropertyGroup):
		def wrapper():
			TRAITBLENDER_CONFIG[cls.__name__] = PointerProperty(type=cls)
		return cls
	return wrapper
return decorator
